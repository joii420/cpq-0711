# 核价管理 — 财务核价工作台（第一期）实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 把「核价管理」菜单改造为财务核价审批节点：报价单提交进入财务角色队列，财务在只读核价工作台复核后整单核价通过/驳回，销售可一步撤回到草稿；不含切版本调价（第二期）。

**Architecture:** 复用现有报价单审批链路 + 详情页只读快照视图。新增精简 `costing_order` 表锚定核价流程；新增 role-based `costingApprove/Reject` 端点（**不**动 `assignedApproverId`，避免连锁现有 8 处）；扩展 `withdraw` 放宽状态 + 解冻并废弃两步撤回；前端复用 `/costing-summary` 路由为新列表 + 新只读核价工作台路由。

**Tech Stack:** Java 17 / Quarkus / Hibernate Panache / Flyway / PostgreSQL；React + Ant Design + TypeScript / Vitest / Playwright。

**Spec:** `docs/superpowers/specs/2026-06-29-核价管理财务核价工作台-design.md`（v3 第一期）。

**纪律：** 开发在隔离 worktree（`superpowers:using-git-worktrees`）。后端改 `.java`/`.sql` 后 touch java 触发 Quarkus 重启自检；前端改后 `tsc --noEmit` + Vite 200；协议级（碰 ReadonlyProductCard 复用渲染）跑双 spec E2E。每个 Task 末 commit，只 `git add` 本次明确文件。

```java
import io.quarkus.hibernate.orm.panache.PanacheEntityBase;
import jakarta.persistence.*;
import java.time.OffsetDateTime;
import java.util.UUID;

@Entity
@Table(name = "costing_order")
public class CostingOrder extends PanacheEntityBase {
    @Id @GeneratedValue public UUID id;

    @Column(name = "quotation_id", nullable = false)
    public UUID quotationId;

    @Column(name = "submitted_by")
    public UUID submittedBy;

    @Column(name = "entered_costing_at", nullable = false)
    public OffsetDateTime enteredCostingAt;

    @Column(name = "created_at", nullable = false)
    public OffsetDateTime createdAt;

    @PrePersist
    public void prePersist() {
        OffsetDateTime now = OffsetDateTime.now();
        if (enteredCostingAt == null) enteredCostingAt = now;
        if (createdAt == null) createdAt = now;
    }

    public static CostingOrder findByQuotation(UUID quotationId) {
        return find("quotationId", quotationId).firstResult();
    }
}
```

```java
    private boolean isFinanceOrAdmin(UUID userId) {
        User u = User.findById(userId);
        return u != null && ("PRICING_MANAGER".equals(u.role) || "SYSTEM_ADMIN".equals(u.role));
    }

    @Transactional
    public QuotationDTO costingApprove(UUID id, String comment, UUID currentUserId) {
        Quotation q = Quotation.findById(id);
        if (q == null) throw new BusinessException(404, "Quotation not found: " + id);
        if (!"SUBMITTED".equals(q.status)) throw new BusinessException(400, "仅待核价(SUBMITTED)可核价通过");
        if (!isFinanceOrAdmin(currentUserId)) throw new BusinessException(403, "仅财务/管理员可核价");
        q.status = "APPROVED";
        writeApproval(id, currentUserId, "COSTING_APPROVED", comment);
        LOG.infof("Costing approved quotation id=%s by=%s", id, currentUserId);
        QuotationDTO dto = QuotationDTO.from(q);
        dto.lineItems = loadLineItems(id);
        return dto;
    }

    @Transactional
    public QuotationDTO costingReject(UUID id, String reason, UUID currentUserId) {
        Quotation q = Quotation.findById(id);
        if (q == null) throw new BusinessException(404, "Quotation not found: " + id);
        if (!"SUBMITTED".equals(q.status)) throw new BusinessException(400, "仅待核价(SUBMITTED)可驳回");
        if (!isFinanceOrAdmin(currentUserId)) throw new BusinessException(403, "仅财务/管理员可核价");
        if (reason == null || reason.isBlank()) throw new BusinessException(400, "驳回原因必填");
        q.status = "COSTING_REJECTED";
        writeApproval(id, currentUserId, "COSTING_REJECTED", reason);
        LOG.infof("Costing rejected quotation id=%s reason=%s by=%s", id, reason, currentUserId);
        QuotationDTO dto = QuotationDTO.from(q);
        dto.lineItems = loadLineItems(id);
        return dto;
    }

    private void writeApproval(UUID quotationId, UUID approverId, String action, String comment) {
        QuotationApproval a = new QuotationApproval();
        a.quotationId = quotationId;
        a.approverId = approverId;
        a.action = action;
        a.comment = comment;
        a.actedAt = OffsetDateTime.now();
        a.persist();
    }
```

```java
    private static String deriveCostingStatus(String quotationStatus) {
        switch (quotationStatus) {
            case "SUBMITTED": return "待核价";
            case "COSTING_REJECTED": return "核价驳回";
            case "APPROVED": case "SENT": case "ACCEPTED": return "核价通过";
            default: return null;   // 其它(DRAFT/REJECTED/EXPIRED/CANCELLED) 不列出
        }
    }

    public java.util.List<CostingOrderListItemDTO> listCostingOrders(String statusFilter, String sort) {
        // costing_order JOIN quotation JOIN user(submitter)
        String jpql = "SELECT co.quotationId, q.quotationNumber, q.snapshotCustomerName, q.baseCurrency, " +
                "u.fullName, q.status, co.enteredCostingAt " +
                "FROM CostingOrder co JOIN Quotation q ON q.id = co.quotationId " +
                "LEFT JOIN User u ON u.id = co.submittedBy " +
                "ORDER BY co.enteredCostingAt DESC";
        java.util.List<Object[]> rows = getEntityManager().createQuery(jpql, Object[].class).getResultList();
        java.util.List<CostingOrderListItemDTO> out = new java.util.ArrayList<>();
        for (Object[] r : rows) {
            String derived = deriveCostingStatus((String) r[5]);
            if (derived == null) continue;                                  // 只列进入过核价的单
            if (statusFilter != null && !statusFilter.isBlank() && !derived.equals(statusFilter)) continue;
            CostingOrderListItemDTO d = new CostingOrderListItemDTO();
            d.quotationId = (UUID) r[0]; d.quotationNumber = (String) r[1];
            d.customerName = (String) r[2]; d.baseCurrency = (String) r[3];
            d.submittedByName = (String) r[4]; d.status = derived;
            d.createdAt = (OffsetDateTime) r[6];
            out.add(d);
        }
        return out;
    }
```

```tsx
import React, { useEffect, useState } from 'react';
import { Tag, Select, Modal, Input, message, Typography } from 'antd';
import { useNavigate } from 'react-router-dom';
import SelectableTable, { runBatch, type ToolbarAction } from '../../components/SelectableTable';
import { costingOrderService, type CostingOrderListItem } from '../../services/costingOrderService';

const STATUS_COLOR: Record<string, string> = { '待核价': 'blue', '核价驳回': 'red', '核价通过': 'green' };
const { Title } = Typography;

const CostingOrderListPage: React.FC = () => {
  const navigate = useNavigate();
  const [data, setData] = useState<CostingOrderListItem[]>([]);
  const [loading, setLoading] = useState(false);
  const [statusFilter, setStatusFilter] = useState<string | undefined>();

  const load = async () => {
    setLoading(true);
    try { const res = await costingOrderService.list({ status: statusFilter }); setData(res.data); }
    finally { setLoading(false); }
  };
  useEffect(() => { load(); }, [statusFilter]);

  const cols = [
    { title: '报价单号', dataIndex: 'quotationNumber', width: 180,
      render: (v: string, r: CostingOrderListItem) =>
        <a onClick={e => { e.stopPropagation(); navigate(`/quotations/${r.quotationId}`); }}>{v}</a> },
    { title: '客户', dataIndex: 'customerName', width: 160 },
    { title: '货币', dataIndex: 'baseCurrency', width: 80 },
    { title: '提交人', dataIndex: 'submittedByName', width: 120 },
    { title: '当前状态', dataIndex: 'status', width: 110,
      render: (s: string) => <Tag color={STATUS_COLOR[s]}>{s}</Tag> },
    { title: '创建时间', dataIndex: 'createdAt', width: 170,
      render: (v: string) => v ? new Date(v).toLocaleString('zh-CN') : '-' },

      } },
    { key: 'reject', label: '驳回', danger: true,
      enabledWhen: rows => rows.length > 0 && rows.every(r => r.status === '待核价') ? true : '仅待核价可驳回',
      onClick: rows => {
        let reason = '';
        Modal.confirm({
          title: `驳回 ${rows.length} 张报价单`,
          content: (<div>
            <div style={{ marginBottom: 8 }}>{rows.map(r => r.quotationNumber).join('、')}</div>
            <Input.TextArea placeholder="驳回原因(必填)" onChange={e => { reason = e.target.value; }} />
          </div>),
          okButtonProps: { danger: true },
          onOk: async () => {
            if (!reason.trim()) { message.error('请填写驳回原因'); throw new Error('reason required'); }
            await runBatch(rows, r => costingOrderService.reject(r.quotationId, reason), '驳回');
            load();
          },
        });
      } },
  ];

  return (
    <div style={{ padding: 16 }}>

        loading={loading} actions={actions} />

};
export default CostingOrderListPage;
```

```tsx
        </Space>}

  );
};
export default CostingReviewPage;
```
