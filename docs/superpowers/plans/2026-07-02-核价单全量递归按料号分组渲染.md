# 核价单「全量递归 + 按料号分组」渲染重构 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 把核价单渲染从「每卡片×每递归组件各跑一次 BOM 闭包 + 逐 occurrence spineKeys 挂数据」重构为「一次可配置递归 SQL 跑全量生产料号 + 每页签 SQL 只跑一次 + 后端按 `material_no` 分组到各卡片」。

**Architecture:** 新增全局可配置递归 SQL 配置表 + `CostingTreeRenderService` 整单编排(1 次递归 + N 页签 + 内存分组),产出 `Map<lineItemId, Map<compId, baseRows>>` 回填现有 `costing_card_values` 快照/公式管线。产品卡片视图与 Excel 视图两条渲染面**共用同一棵树**(同一 `__nodeId` 真理源)。彻底删除旧 `BomClosureService`/spineKeys 引擎。硬切换、不迁移存量。

**Tech Stack:** Java 17 + Quarkus 3.23 + Hibernate Panache + Flyway + PostgreSQL 16(`WITH RECURSIVE`);React + Ant Design(配置页签);Playwright(E2E)。

**Spec:** `docs/superpowers/specs/2026-07-02-核价单全量递归按料号分组渲染-design.md`(实现须逐条对齐 §14 验收清单)。

package com.cpq.component.service;

import jakarta.enterprise.context.ApplicationScoped;
import jakarta.inject.Inject;
import javax.sql.DataSource;
import java.sql.*;
import java.util.*;

/** 核价树递归 SQL 保存期 dry-run 校验:变量存在 + 4 约定列 + 可执行(空 seed)。 */
@ApplicationScoped
public class CostingTreeSqlValidator {
    @Inject DataSource dataSource;

    public static final List<String> REQUIRED_COLS = List.of("root_no", "material_no", "bom_version", "parent_no");

    public static final class Result {
        public final boolean ok; public final String message;
        Result(boolean ok, String message) { this.ok = ok; this.message = message; }
    }

    public Result validate(String sql) {
        if (sql == null || sql.isBlank()) return new Result(false, "SQL 不能为空");
        if (!sql.contains(":production_part_nos"))
            return new Result(false, "递归 SQL 必须引用 :production_part_nos");
        // 用空 text[] 占位执行一次,LIMIT 0 只校验列结构,不实际爬树
        String probe = "SELECT * FROM (" + sql.replace(":production_part_nos", "ARRAY[]::text[]") + ") q LIMIT 0";
        try (Connection c = dataSource.getConnection();
             PreparedStatement ps = c.prepareStatement(probe);
             ResultSet rs = ps.executeQuery()) {
            ResultSetMetaData meta = rs.getMetaData();
            Set<String> cols = new HashSet<>();
            for (int i = 1; i <= meta.getColumnCount(); i++) cols.add(meta.getColumnLabel(i).toLowerCase());
            for (String need : REQUIRED_COLS)
                if (!cols.contains(need)) return new Result(false, "递归 SQL 缺输出列: " + need);
            return new Result(true, "ok");
        } catch (Exception e) {
            return new Result(false, "递归 SQL 无法执行: " + e.getMessage());
        }
    }
}
```

package com.cpq.component.service;

import com.cpq.component.entity.CostingBomTreeConfig;
import jakarta.enterprise.context.ApplicationScoped;
import jakarta.inject.Inject;
import jakarta.transaction.Transactional;
import java.util.List;
import java.util.UUID;

@ApplicationScoped
public class CostingBomTreeConfigService {
    @Inject CostingTreeSqlValidator validator;

    public List<CostingBomTreeConfig> list() {
        return CostingBomTreeConfig.listAll();
    }

    @Transactional
    public CostingBomTreeConfig create(String name, String sqlTemplate) {
        var r = validator.validate(sqlTemplate);
        if (!r.ok) throw new RuntimeException("递归 SQL 校验失败: " + r.message);
        var e = new CostingBomTreeConfig();
        e.name = name; e.sqlTemplate = sqlTemplate; e.isActive = false;
        e.persist();
        return e;
    }

    @Transactional
    public CostingBomTreeConfig update(UUID id, String name, String sqlTemplate) {
        var r = validator.validate(sqlTemplate);
        if (!r.ok) throw new RuntimeException("递归 SQL 校验失败: " + r.message);
        CostingBomTreeConfig e = CostingBomTreeConfig.findById(id);
        if (e == null) throw new RuntimeException("配置不存在: " + id);
        e.name = name; e.sqlTemplate = sqlTemplate;
        return e;
    }

    /** 设为生效:单事务先把当前 active 置 false(避开部分唯一索引),再置目标 true。 */
    @Transactional
    public void setActive(UUID id) {
        CostingBomTreeConfig target = CostingBomTreeConfig.findById(id);
        if (target == null) throw new RuntimeException("配置不存在: " + id);
        CostingBomTreeConfig.update("isActive = false where isActive = true");
        CostingBomTreeConfig.getEntityManager().flush();   // 先落 false 再置 true,防唯一索引瞬时冲突
        target.isActive = true;
    }

    @Transactional
    public void delete(UUID id) {
        CostingBomTreeConfig.deleteById(id);
    }
}
```

    public static void set(Vars v) { TL.set(v); }
    public static Vars get() { return TL.get(); }
    public static void clear() { TL.remove(); }
    private CostingTreeVarsContext() {}
}

```java
package com.cpq.quotation.service;

import java.util.*;

/** 递归行 → 分组产物 + 树 nodeId 生成(纯函数,无 DB)。 */
public final class CostingTreeGrouping {
    private CostingTreeGrouping() {}

    public static final class Result {
        public final Map<String, LinkedHashSet<String>> cardMaterialNo;   // root_no → distinct material_no(含根)
        public final List<String> totalMaterialNo;                        // distinct 全量
        public final Map<String, List<CostingTreeNode>> treeRowsByRoot;   // root_no → 保序树行(含 nodeId/parentId/lvl)
        Result(Map<String, LinkedHashSet<String>> c, List<String> t, Map<String, List<CostingTreeNode>> tr) {
            this.cardMaterialNo = c; this.totalMaterialNo = t; this.treeRowsByRoot = tr;
        }
    }

    public static Result group(List<CostingTreeNode> rows) {
        Map<String, LinkedHashSet<String>> cardMat = new LinkedHashMap<>();
        LinkedHashSet<String> total = new LinkedHashSet<>();
        Map<String, List<CostingTreeNode>> byRoot = new LinkedHashMap<>();

        for (CostingTreeNode r : rows) {
            cardMat.computeIfAbsent(r.rootNo, k -> new LinkedHashSet<>()).add(r.materialNo);
            total.add(r.materialNo);
            byRoot.computeIfAbsent(r.rootNo, k -> new ArrayList<>()).add(r);
        }

        // 每棵树内生成 nodeId/parentId/lvl。nodeId = root_no + '#' + 出现序(保 occurrence 唯一);
        // parentId 由 (root, parent_no) 映射到父节点的 nodeId;同 parent_no 多 occurrence 时取"最近未消费"父,
        // 简化:同 root 内按出现序,parentId = 该 root 下 materialNo==parent_no 的首个节点 nodeId(DAG 摊平,占位序号保唯一)。
        for (Map.Entry<String, List<CostingTreeNode>> e : byRoot.entrySet()) {
            String root = e.getKey();
            List<CostingTreeNode> tree = e.getValue();
            Map<String, String> firstNodeIdOfMat = new HashMap<>();  // materialNo → 首个 nodeId(建 parentId 用)
            int seq = 0;
            for (CostingTreeNode nd : tree) {
                nd.nodeId = root + "#" + (seq++);
                firstNodeIdOfMat.putIfAbsent(nd.materialNo, nd.nodeId);
            }
            for (CostingTreeNode nd : tree) {
                nd.parentId = (nd.parentNo == null) ? null : firstNodeIdOfMat.get(nd.parentNo);
                nd.lvl = (nd.parentNo == null) ? 1 : 0;  // lvl 在下一步按 parentId 链回填
            }
            // 按 parentId 回填 lvl(根=1,子=父+1;DAG 摊平后按首父深度)
            Map<String, CostingTreeNode> byId = new HashMap<>();
            for (CostingTreeNode nd : tree) byId.put(nd.nodeId, nd);
            for (CostingTreeNode nd : tree) nd.lvl = depth(nd, byId, new HashSet<>());
        }
        return new Result(cardMat, new ArrayList<>(total), byRoot);
    }

    private static int depth(CostingTreeNode nd, Map<String, CostingTreeNode> byId, Set<String> seen) {
        if (nd.parentId == null || !seen.add(nd.nodeId)) return 1;
        CostingTreeNode p = byId.get(nd.parentId);
        return p == null ? 1 : 1 + depth(p, byId, seen);
    }
}
```
> **注(实现者判断点)**:上面的 parentId「首个同料号节点」是 DAG 摊平的简化。若配置员的递归 SQL 已能唯一定位每条边的父节点(理想),更稳的做法是让递归 SQL 额外输出边序号并据此配对。实现时若发现层级错乱,回本 spec §7.1 与用户确认 nodeId 生成口径,不要擅自改契约。

```java
package com.cpq.quotation.service;

import com.cpq.component.entity.CostingBomTreeConfig;
import com.cpq.datasource.sqlview.CostingTreeVarsContext;
import com.cpq.common.BusinessException;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.node.*;
import jakarta.enterprise.context.ApplicationScoped;
import jakarta.inject.Inject;
import javax.sql.DataSource;
import java.sql.*;
import java.util.*;

@ApplicationScoped
public class CostingTreeRenderService {
    private static final ObjectMapper MAPPER = new ObjectMapper();
    @Inject DataSource dataSource;
    // 复用现有跑组件 $view 的执行入口(参照 ComponentDriverService.expandForPartSet 内部走的 SqlViewExecutor 路径);
    // 实现时用 codegraph 定位「按组件跑 $view 返回行」的最小 API,注入之。
    // @Inject ComponentDriverService componentDriverService;

    /** 整单编排产物:Map<lineItemId, Map<compId(:sortOrder), baseRows>>。 */
    public Map<UUID, Map<String, ArrayNode>> renderQuotation(
            List<QuotationLineItem> lineItems, java.util.UUID templateId, java.util.UUID customerId) {
        // 1. seed
        LinkedHashSet<String> seed = new LinkedHashSet<>();
        for (var li : lineItems) if (li.productPartNoSnapshot != null && !li.productPartNoSnapshot.isBlank())
            seed.add(li.productPartNoSnapshot);
        if (seed.isEmpty()) return Map.of();

        // 2. 递归
        CostingBomTreeConfig cfg = CostingBomTreeConfig.findActive();
        if (cfg == null) throw new BusinessException(400, "未配置生效的核价树递归 SQL");
        List<CostingTreeNode> treeRows = runRecursive(cfg.sqlTemplate, new ArrayList<>(seed));

        // 3. 分组
        CostingTreeGrouping.Result g = CostingTreeGrouping.group(treeRows);

        // 4. 跑每个页签 $view 一次(:total_material_no) → rowsByMaterial(按 compId)
        //    driverComps = 模板里有 data_driver_path 的组件 [{compId, isTreeTab(bom_recursive_expand)}]
        //    (SQL 与 CardSnapshotService.expandTemplateDriverBaseRows 里那句 DISTINCT 一致)
        //    treeTabCompId = 唯一 bom_recursive_expand=true 的组件(§11 校验保证 ≤1)
        // ... 见 Step 2/3

        // 5/6. 逐卡逐页签装 baseRows(见 Step 3)
        return Map.of(); // 占位,Step 3 补全
    }

    /** 跑递归 SQL,映射 4 列 → CostingTreeNode。 */
    private List<CostingTreeNode> runRecursive(String sql, List<String> seed) {
        List<CostingTreeNode> out = new ArrayList<>();
        CostingTreeVarsContext.set(new CostingTreeVarsContext.Vars(seed, null));
        try (Connection c = dataSource.getConnection()) {
            String prepared = sql.replace(":production_part_nos", "?");
            try (PreparedStatement ps = c.prepareStatement(
                    "SELECT root_no, material_no, bom_version, parent_no FROM (" + prepared + ") q")) {
                ps.setArray(1, c.createArrayOf("text", seed.toArray(new String[0])));
                try (ResultSet rs = ps.executeQuery()) {
                    while (rs.next()) out.add(new CostingTreeNode(
                        rs.getString("root_no"), rs.getString("material_no"),
                        rs.getString("bom_version"), rs.getString("parent_no")));
                }
            }
        } catch (Exception e) {
            throw new BusinessException(500, "核价树递归 SQL 执行失败: " + e.getMessage());
        } finally {
            CostingTreeVarsContext.clear();
        }
        return out;
    }
}
```

        if (bizRow != null) bizRow.forEach((k,v) -> driver.set(k, MAPPER.valueToTree(v)));

        driver.put("__nodeId", node.nodeId);
        driver.put("__parentId", node.parentId);
        driver.put("__lvl", node.lvl);
        row.putObject("basicDataValues");
        return row;
    }

    static ObjectNode flatRowNode(Map<String,Object> bizRow) {
        ObjectNode row = MAPPER.createObjectNode();

        bizRow.forEach((k,v) -> driver.set(k, MAPPER.valueToTree(v)));
        row.putObject("basicDataValues");
        return row;
