```java
package com.cpq.basicdata.v6.versioning;

import io.quarkus.test.junit.QuarkusTest;
import jakarta.inject.Inject;
import jakarta.persistence.EntityManager;
import jakarta.transaction.Transactional;
import org.junit.jupiter.api.*;
import java.util.*;
import static org.junit.jupiter.api.Assertions.*;

@QuarkusTest
class VersionedV6SortKeyTest {
    @Inject VersionedV6Writer writer;
    @Inject EntityManager em;
    static final String FMN = "TEST-SORT-0001";

    @AfterEach @Transactional
    void clean() { em.createNativeQuery("DELETE FROM unit_price WHERE finished_material_no=:f")
        .setParameter("f", FMN).executeUpdate(); }

    private VersionedGroupSpec spec(List<Map<String,Object>> rows) {
        return new VersionedGroupSpec("unit_price","version_no",
            new LinkedHashMap<>(Map.of("system_type","QUOTE","customer_no","C1",
                "price_type","MATERIAL","cost_type","来料其他费用","finished_material_no",FMN,"code",FMN)),
            // ⚠️ 首列 cost_ratio 组内重复（考验 multiset vs 首列排序）；行靠 seq_no 区分。
            // 关键：seq_no 在 uq_unit_price 内、cost_ratio 不在 → 两行可共存不撞键。
            // 若反过来（同 seq_no 差 cost_ratio）第一次 INSERT 即撞 uq_unit_price（实测踩坑）。
            List.of("cost_ratio","seq_no"), rows);
    }
    private Map<String,Object> r(String ratio, int seq){ Map<String,Object> m=new LinkedHashMap<>();
        m.put("cost_ratio",new java.math.BigDecimal(ratio)); m.put("seq_no",seq); return m; }

    @Test @Transactional
    void sameContentDifferentOrder_sameFirstCol_reusesVersion() {
        writer.writeVersionedGroup(spec(List.of(r("10",1), r("10",2))));           // 首列同为 10，靠 seq_no 区分
        String v2 = writer.writeVersionedGroup(spec(List.of(r("10",2), r("10",1)))); // 顺序颠倒、内容相同
        assertEquals("2000", v2, "同首列多行、仅顺序不同应判'相同'不升版（multiset 行为）");
        Number total = (Number) em.createNativeQuery(
            "SELECT count(*) FROM unit_price WHERE finished_material_no=:f").setParameter("f",FMN).getSingleResult();
        assertEquals(2L, total.longValue(), "不应凭空升版翻倍");
    }
}
```

```java
/**
 * BOM 主从版本化（设计 §3.3 / §5.2 / §5.3）。
 * 子表行集决定是否升版；版本号取主表版本列 max+1；主/子 is_current 同步翻转。
 * - childVersionColumn != null（element_bom_item，uq 含 characteristic）：子表写版本列 → 多版本子表保留。
 * - childVersionColumn == null（material_bom_item，uq 不含版本）：子表走 upsert 覆盖 → 仅保留当前版本（#7）。
 * - 主/子 groupKey 独立（#5：material_bom_item 无 bom_type 列）。
 * - 主表固定列 masterFixedColumns（#4：element_bom/material_bom 的 bom_type NOT NULL）。
 * - 所有组匹配 NULL 安全（#8：material_bom 子表 characteristic=NULL）。
 *
 * @param masterTable / masterVersionColumn   主表名 + 版本列（element_bom→characteristic, material_bom→bom_version）
 * @param masterGroupKey                      主表身份（含 bom_type/bom_version 维度），写主表 + 定位现有主组
 * @param masterFixedColumns                  主表 NOT NULL 固定列（如 {bom_type:'MATERIAL'}）；可空 Map
 * @param childTable / childVersionColumn     子表名 + 子版本列（null=子表不写版本列，走 upsert）
 * @param childGroupKey                       子表身份（仅子表真实存在的列；含 characteristic 判别）
 * @param childContentColumns                 子表指纹列 + 写库值列
 * @param childRows                           本组子表整套行
 * @return 本组版本号
 */
public String writeVersionedMasterDetail(
        String masterTable, String masterVersionColumn,
        Map<String,Object> masterGroupKey, Map<String,Object> masterFixedColumns,
        String childTable, String childVersionColumn,
        Map<String,Object> childGroupKey, List<String> childContentColumns,
        List<Map<String,Object>> childRows) {

    if (!ALLOWED_TABLES.contains(masterTable) || !ALLOWED_TABLES.contains(childTable))
        throw new IllegalArgumentException("表未登记白名单");
    safeIdent(masterTable); safeIdent(masterVersionColumn); safeIdent(childTable);
    if (childVersionColumn != null) safeIdent(childVersionColumn);
    masterGroupKey.keySet().forEach(VersionedV6Writer::safeIdent);
    childGroupKey.keySet().forEach(VersionedV6Writer::safeIdent);
    childContentColumns.forEach(VersionedV6Writer::safeIdent);
    if (masterFixedColumns != null) masterFixedColumns.keySet().forEach(VersionedV6Writer::safeIdent);
    if (childRows.isEmpty()) throw new IllegalArgumentException("childRows 为空;整组下线请用专门 API");
    if (childContentColumns.isEmpty()) throw new IllegalArgumentException("childContentColumns 不能为空");

    // 并发串行化（advisory lock，复用 writeVersionedGroup 同款）
    advisoryLock(masterTable, masterGroupKey);

    // 1. 比对子表当前生效组（NULL 安全 where + multiset 比较，复用现有 tally/norm）
    List<Map<String,Object>> existingChild = loadCurrentChild(childTable, childGroupKey, childContentColumns);
    if (multisetEqual(existingChild, childRows, childContentColumns)) {
        return currentVersionOf(masterTable, masterVersionColumn, masterGroupKey);  // 完全不写库（决策①）
    }

    // 2. 升版号 = 主表版本列全历史 max+1（V1 等非数字被正则忽略，首版 2000）
    String newVersion = nextVersionOf(masterTable, masterVersionColumn, masterGroupKey);

    // 3. 主 + 子旧组下线（NULL 安全）
    flip(masterTable, masterGroupKey);
    flip(childTable, childGroupKey);

    // 4. 写主表一行（groupKey + 固定列 + 版本 + is_current）
    Map<String,Object> master = new LinkedHashMap<>(masterGroupKey);
    if (masterFixedColumns != null) master.putAll(masterFixedColumns);
    master.put(masterVersionColumn, newVersion);
    master.put("is_current", true);
    insertRowGeneric(masterTable, master);

    // 5. 写子表行集
    List<String> childUq = childUniqueCols(childTable);   // 从已知 uq 定义取（见下）
    for (Map<String,Object> row : childRows) {
        Map<String,Object> all = new LinkedHashMap<>(childGroupKey);
        all.putAll(row);
        all.put("is_current", true);
        if (childVersionColumn != null) {                 // element_bom_item：写版本列 → 多版本保留
            all.put(childVersionColumn, newVersion);
            insertRowGeneric(childTable, all);
        } else {                                          // material_bom_item：uq 无版本 → upsert 覆盖(#7)
            upsertChildRow(childTable, childUq, all);
        }
    }
    return newVersion;
}
```
> 配套实现要点：
> - `loadCurrentChild` / `currentVersionOf` / `multisetEqual` 直接复用现有 `loadCurrentGroup`/`currentVersion`/`tally`+`norm` 的逻辑，仅把入参从 `VersionedGroupSpec` 改为显式 `(table, groupKey, contentCols)`。**不新增比对/升版算法。**
> 🔧 **实现实测回写（2026-06-01，最终落地与上方伪代码的差异，以本节为准）：**
> - **`ON CONFLICT` 用表达式目标，不能用 `ON CONSTRAINT 索引名`**：`uq_element_bom_item` / `uq_material_bom_item` 都是 `CREATE UNIQUE INDEX`（**表达式索引，含 `COALESCE`**），`ON CONFLICT ON CONSTRAINT` 对表达式索引无效。落地用静态登记 `CHILD_UQ`（`record ChildUq(String conflictTarget, Set<String> keyCols)`），material_bom_item 的 `conflictTarget = "(system_type, customer_no, material_no, COALESCE(characteristic,''), COALESCE(seq_no,0), COALESCE(component_no,''), COALESCE(part_no,''))"`（精确复刻 V219 + 现有 Q03 写法）。
> - **`upsertChildRow` 的 `DO UPDATE SET` 只更"非 keyCols"列**（keyCols 来自 `CHILD_UQ`），含 `is_current = EXCLUDED.is_current`（恒为 true）。未刷 `updated_at`（`all` 不含该列；如需可后补，不影响版本语义）。
> - **✅ 残留子行改为 DELETE 清理（比 V2.0「接受残留」更干净）**：第 6 步对 `childVersionColumn==null` 路径执行 `deleteNonCurrent(childTable, childGroupKey)`（`DELETE ... WHERE <childGroupKey NULL安全> AND is_current=FALSE`）。原因：改了 `component_no` 的旧子行（uq 含 component_no，新行不覆盖它）会以 is_current=false 残留 → DELETE 后 material_bom_item 真正"仅当前版本明细"（§5.3）。**element_bom_item（childVersionColumn!=null）绝不走此步**，否则删除多版本历史。
> - **`masterFixedColumns` 形参类型 = `Map<String, ?>`**（兼容 `Map.of()` 空表 → `Map<Object,Object>` 与 `Map.of("k","v")` → `Map<String,String>`；统一 putAll 进 `Map<String,Object>`）。
> - **`loadCurrentChild` 即复用后的 `loadCurrentGroup(table, groupKey, contentCols)`**（已参数化，单表/主从共用，无 ORDER BY，multiset 比较顺序无关）。
